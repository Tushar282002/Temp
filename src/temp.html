Assigned To: {{ user }}

{% if message %}

{{ message }}

{% endif %} """ @app.route('/edit', methods=['GET', 'POST']) def edit(): task_id = request.args.get('task_id', 'N/A') user = request.args.get('user', 'N/A') # Get current system date and default time now = datetime.now() default_date = now.strftime('%Y-%m-%d') # e.g. 2025-07-03 default_time = "00:00" # for 00H:00M message = None if request.method == 'POST': date = request.form.get('date') time = request.form.get('time') message = f"Received Date: {date}, Time: {time} for Task ID {task_id} (not saved, DB part pending)." return render_template_string(FORM_HTML, task_id=task_id, user=user, default_date=default_date, default_time=default_time, message=message) if __name__ == '__main__': app.run(host="0.0.0.0", port=5000)
